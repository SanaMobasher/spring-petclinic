 package org.springframework.samples.petclinic.system;
 
 import org.springframework.stereotype.Controller;
 import org.springframework.web.bind.annotation.GetMapping;
 
 /**
  * Controller used to showcase what happens when an exception is thrown
  *
  * <p>Author: Michael Isvy</p>
  *
  * Also see how a view that resolves to "error" has been added ("error.html").
  */
 
 // Custom exception to replace RuntimeException
 class CustomControllerException extends RuntimeException {
 	public CustomControllerException(String message) {
 		super(message);
 	}
 }
 
 @Controller
 public class CrashController {
 
 	@GetMapping("/oups")
 	public String triggerException() {
 		// Throwing a dedicated exception instead of RuntimeException
 		throw new CustomControllerException(
 			"Expected: controller used to showcase what happens when an exception is thrown");
 	}
 }
